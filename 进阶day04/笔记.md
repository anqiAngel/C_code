# 字符串函数

## string.h库

1.strlen(首字符地址)
>1.从首字符开始数数到'\0'停止 返回'\0'前面出现的字符个数(不包含'\0')
>2.参数指向的字符串必须要以'\0'结束
>3.函数返回值是无符号的

    int main()
    {
        /* code */
        char str[] = {'A', 'B', 'C'};
        printf("%d\n", strlen(str)); // 随机值
        // strlen()的返回值是unsigned long long类型 unsigned运算还是unsigned 表达式永远 > 0 -> 输出>
        if (strlen("abc") - strlen("abcdef") > 0)
        {
            /* code */
            printf(">\n");
        }
        else
        {
            /* code */
            printf("<=\n");
        }
    }
